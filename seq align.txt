
#include <stdio.h>

void main() 
	 
{ 
	FILE *fopen(), *fp; 
	int c , nc, nlines; 
	char filename[40] ;

	nlines = 0 ;
	nc = 0; 

	printf(“Enter file name: “); 
	gets( filename ); 

	fp = fopen( filename, “r” );     

	if ( fp == NULL ) 
	{ 
		printf(“Cannot open %s for reading \n”, filename ); 
		exit(1);      /* terminate program */ 
	} 

	c = getc( fp ) ;	        
	while (  c != EOF ) 
	{ 
		if ( c  ==  ‘\n’  )	  
			nlines++ ;

		nc++ ;  
		c = getc ( fp ); 
	} 

	fclose( fp ); 

	
}
